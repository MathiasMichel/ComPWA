84  // lets define a modulus operator for our special comparison
85  // we want to check if all the properties of rhs are available in this
86  bool supersetOf(const SpinWave &rhs) const {
87  for (auto const &spin_like_qn : rhs.spin_like_quantum_numbers_) {
88  bool nothing_found(true);
89  auto result = spin_like_quantum_numbers_.find(spin_like_qn.first);
90  if (result != spin_like_quantum_numbers_.end()) {
91  if (1.0 * result->second.GetNumerator() /
92  result->second.GetDenominator() ==
93  1.0 * spin_like_qn.second.GetNumerator() /
94  spin_like_qn.second.GetDenominator()) {
95  if (result->second.UseZ()) {
96  if (1.0 * result->second.GetZNumerator() /
97  result->second.GetDenominator() ==
98  1.0 * spin_like_qn.second.GetZNumerator() /
99  spin_like_qn.second.GetDenominator()) {
100  nothing_found = false;
101  }
102  } else
103  nothing_found = false;
104  }
105  }
106  if (nothing_found)
107  return false;
108  }
109  for (auto const &int_like_qn : rhs.integer_like_quantum_numbers_) {
110  auto result = integer_like_quantum_numbers_.find(int_like_qn.first);
111  if (result != integer_like_quantum_numbers_.end()) {
112  if (result->second != int_like_qn.second) {
113  return false;
114  }
115  }
116  }
117  for (auto const &double_like_qn : rhs.double_like_quantum_numbers_) {
118  auto result = double_like_quantum_numbers_.find(double_like_qn.first);
119  if (result != double_like_quantum_numbers_.end()) {
120  if (result->second != double_like_qn.second) {
121  return false;
122  }
123  }
124  }
125 
126  return true;
127  }
