110  void addMappingIfUnique(const IndexMapping& mapping) {
111  bool add_mapping(true);
112 
113  // loop over current mappings
114  for (unsigned int i = 0; i < current_unique_mappings_.size(); ++i) {
115  // now for each final state content list compare the outcome
116  // of the current mapping and the reference
117  bool mappings_identical(true);
118  for (auto final_state_content_id_list = unique_id_lists_.begin();
119  final_state_content_id_list != unique_id_lists_.end();
120  ++final_state_content_id_list) {
121 
122  if (createFinalStateMappedList(*final_state_content_id_list,
123  current_unique_mappings_[i])
124  != createFinalStateMappedList(*final_state_content_id_list,
125  mapping)) {
126  mappings_identical = false;
127  break;
128  }
129  }
130  if (!mappings_identical) {
131  break;
132  }
133  else {
134  add_mapping = false;
135  break;
136  }
137  }
138 
139  if (add_mapping)
140  current_unique_mappings_.push_back(mapping);
141  }
