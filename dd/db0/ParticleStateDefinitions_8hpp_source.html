159  bool operator()(const std::vector<ComPWA::IDInfo>& lhs,
160  const std::vector<ComPWA::IDInfo>& rhs) const {
161  if (lhs.size() < rhs.size())

164  return false;

169  ComPWA::IDInfo::lessThenIgnoringID);

171  ComPWA::IDInfo::lessThenIgnoringID);

173  if (clone_lhs[i].particle_id_ < clone_rhs[i].particle_id_)
174  return true;
