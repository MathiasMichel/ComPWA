135 inline SubSystem SubSystemFactory(const boost::property_tree::ptree pt) {
136  // Read subSystem definition
137  std::vector<int> recoilState;
138  auto recoil =
139  pt.get_optional<std::string>("RecoilSystem.<xmlattr>.FinalState");
140  if (recoil) {

143 
144  auto decayProducts = pt.get_child("DecayProducts");
145  std::vector<std::vector<int>> finalStates;
146  std::vector<int> helicities;
147  std::vector<std::string> finalStatesNames;
148  for (auto i : decayProducts) {
149  finalStates.push_back(
150  stringToVectInt(i.second.get<std::string>("<xmlattr>.FinalState")));
151  finalStatesNames.push_back(i.second.get<std::string>("<xmlattr>.Name"));
152  helicities.push_back(i.second.get<int>("<xmlattr>.Helicity"));
153  }
154  SubSystem subSys(recoilState, finalStates);
155  subSys.SetFinalStatesNames(finalStatesNames);
156  subSys.SetHelicities(helicities);
157  return subSys;
158 }
