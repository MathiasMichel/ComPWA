223  void SetCouplings(std::vector<Coupling> vC) {
224  if (vC.size() != 2 && vC.size() != 3)
225  throw std::runtime_error(
226  "AmpFlatteRes::SetCouplings() | Vector with "
227  "couplings has a wrong size. We expect either 2 or 3 couplings.");
228 
229  _g = vC;
230 
231  if(_g.size() == 2) _g.push_back(Coupling());
232  // Check if one of the coupling match the final state (_daughterMasses)
233  auto mm = GetDecayMasses();
234  if (mm == std::pair<double, double>(-999, -999))
235  LOG(info)
236  << "AmpFlatteRes::SetCouplings() | Masses of decay products not set. "
237  " Can not determine if correct couplings were set.";
238 
239  bool ok = false;
240  for (auto i : _g) {
241  if (i.GetMassA() == mm.first && i.GetMassB() == mm.second)
242  ok = true;
243  if (i.GetMassB() == mm.first && i.GetMassA() == mm.second)
244  ok = true;
245  }
246  if (!ok)
247  throw std::runtime_error("AmpFlatteRes::SetCouplings() | No couplings "
248  "for the current decay particles set!");
249  }
