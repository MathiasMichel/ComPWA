49  virtual double Intensity(const ComPWA::dataPoint &point) const {
50 
51  // We have to get around the constness of the interface definition.
52  std::vector<std::vector<double>> parameters(_parameters);
53  std::vector<double> normValues(_normValues);
54  if(_intens.size() != parameters.size() )
55  parameters = std::vector<std::vector<double>>(_intens.size());
56  if(_intens.size() != normValues.size() )
57  normValues = std::vector<double>(_intens.size());
58 
59  double result = 0;
60  for(int i=0; i< _intens.size(); i++){
61  std::vector<double> params;
62  _intens.at(i)->GetParametersFast(params);
63  if ( parameters.at(i) != params ) { //recalculate normalization
64  parameters.at(i) = params;
65  normValues.at(i) = 1/Tools::Integral(_intens.at(i), _phspSample);
66  normValues.at(i) *= _intens.at(i)->GetStrength();
67  }
68  result += _intens.at(i)->Intensity(point) * normValues.at(i);
69  }
70 
71  const_cast<std::vector<std::vector<double>>&>(_parameters) = parameters;
72  const_cast<std::vector<double>&>(_normValues) = normValues;
73 
74  return (GetStrength() * result);
75  }
