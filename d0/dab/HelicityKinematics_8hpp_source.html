59  virtual std::string to_string() const {
60  // Creating unique title
61  std::stringstream stream;
62  stream << "(";
63  for (auto i : _recoilState)
64  stream << std::to_string(i);
65  stream << ")->(";
66  for (auto j=_finalStates.begin(); j != _finalStates.end(); ++j) {
67  for (auto i : *j)
68  stream << std::to_string(i);
69  if( j != _finalStates.end()-1 ) stream << ")+(";
70  else stream << ")";
71  }
72 
73  return stream.str();
74  }
