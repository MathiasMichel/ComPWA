290 inline gsl_vector *gsl_parameterList2Vec(const ParameterList &list) {
291  gsl_vector *tmp = gsl_vector_alloc(list.GetNDouble());
292  unsigned int t = 0;
293  for (unsigned int o = 0; o < list.GetNDouble(); o++) {
294  std::shared_ptr<DoubleParameter> outPar = list.GetDoubleParameter(o);
295  if (outPar->IsFixed())
296  continue;
297  gsl_vector_set(tmp, t, outPar->GetValue());
298  t++;
299  }
300  // resize vector
301  gsl_vector *vec = gsl_vector_alloc(t);
302  for (unsigned int i = 0; i < vec->size; i++)
303  gsl_vector_set(vec, i, gsl_vector_get(tmp, i));
304  return vec;
305 };

328 inline void multivariateGaussian(const gsl_rng *rnd, const int vecSize,
329  const gsl_vector *in, const gsl_matrix *cov,
330  gsl_vector *res) {
331  // Generate and fill temporary covariance matrix
332  gsl_matrix *tmpM = gsl_matrix_alloc(vecSize, vecSize);
333  gsl_matrix_memcpy(tmpM, cov);
334 
335  // Cholesky decomposition
336  int status = gsl_linalg_cholesky_decomp(tmpM);
337  if (status == GSL_EDOM)
338  LOG(error) << "Decomposition has failed!";
339 
340  // Compute vector of random gaussian variables
341  for (unsigned int i = 0; i < vecSize; i++)
342  gsl_vector_set(res, i, gsl_ran_ugaussian(rnd));
343 
344  // Debug
345  // gsl_matrix_print(cov);
346  // gsl_vector_print(in);
347  // gsl_vector_print(res);
348  // gsl_matrix_print(tmpM);
349 
350  /*From the GNU GSL Documentation:
351  * The function dtrmv compute the matrix-vector product x = op(A) x for the
352  * triangular matrix A, where op(A) = A, A^T, A^H for TransA = CblasNoTrans,
353  * CblasTrans, CblasConjTrans. When Uplo is CblasUpper then the upper
354  * triangle of A is used, and when Uplo is CblasLower then the lower
355  * triangle of A is used. If Diag is CblasNonUnit then the diagonal of
356  * the matrix is used, but if Diag is CblasUnit then the diagonal elements
357  * of the matrix A are taken as unity and are not referenced.
358  */
359  gsl_blas_dtrmv(CblasLower, CblasNoTrans, CblasNonUnit, tmpM, res);
360 
361  gsl_vector_add(res, in);
362  // free temporary object
363  gsl_matrix_free(tmpM);
364 };
