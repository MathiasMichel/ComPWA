231 inline std::complex<double> couplingToWidth(double mSq, double mR, double g,
232  double ma, double mb, double spin,
233  double mesonRadius,
234  formFactorType type,
235  std::complex<double> phspFactor) {
236  // calculate gammaA(s_R)
237  std::complex<double> gammaA(1, 0); // spin==0
238 
239  if (spin > 0 || type == formFactorType::CrystalBarrel) {
240  std::complex<double> qValue = Kinematics::qValue(mR, ma, mb);
241  double ffR = HelicityKinematics::FormFactor(mR, ma, mb, spin, mesonRadius,
242  qValue, type);
243  std::complex<double> qR = std::pow(qValue, spin);
244  gammaA = ffR * qR;
245  }
246 
247  // calculate phsp factor
248  std::complex<double> res = std::norm(gammaA) * g * g * phspFactor / mR;
249 
250 #ifndef NDEBUG
251  // check for NaN
252  if (std::isnan(res.real()) || std::isnan(res.imag()))
253  throw std::runtime_error("AmpAbsDynamicalFunction::couplingToWidth() | "
254  "Result is NaN!");
255  // check for inf
256  if (std::isinf(res.real()) || std::isinf(res.imag()))
257  throw std::runtime_error("AmpAbsDynamicalFunction::couplingToWidth() | "
258  "Result is inf!");
259 #endif
260 
261  return res;
262 }
